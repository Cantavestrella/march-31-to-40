\begin{estrofa}

\numero{1} \textparagraph\  Lome pelmon no muntan gran valer

\numero{2} Sens hauer bens bondat linatge gran

\numero{3} Mas la delmig valmes quela restan

\numero{4} E no valmolt sens les altres hauer

\numero{5} Per ellas fan les dues molt prear

\numero{6} Car poder val tant com es ministrat

\numero{7} Linatge val aytant com es honrat

\numero{8} La valor dom ho fa tot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no sera lom sabent de sonar

\numero{10} Sin algun teps no sona estrumen

\numero{11} Car per voler sonar lo nom nos pren

\numero{12} Mas lestrument sonant ben acordar

\numero{13} Tot enaxi aquell que dins si val

\numero{14} Pobre de bens e dauilat linatge

\numero{15} No tels areus per mostrar gran coratge

\numero{16} En la virtut ques nomena moral

\end{estrofa}



\begin{estrofa}

\numero{17} [63] Son eseran molts dun altre cabal

\numero{18} Auents molts bens edalta sanch fauor

\numero{19} E valen poch enlapart no mellor

\numero{20} Car sens lom bo quant pot hauer es mal

\numero{21} E montes ve\textit{us} anlacolpa natura

\numero{22} Car fora bo tal que valer no sab

\numero{23} Negun saber no pot viuren son cap

\numero{24} Sens culpa lur de valer han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels estrems virtut almig atura

\numero{26} Molt greu dobrar hi entre pochs conegut

\numero{27} Per ell saber nos hom perbo tengut

\numero{28} Mas fets hobrant forans dins tal mesura

\numero{29} Aytant es larch lom meyns de fer larguessa

\numero{30} Com es estas sino fall endespendre

\numero{31} Vicis virtuts per actes deuen pendre

\numero{32} Apres lo fet es lur potenca pressa

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conquerran virtuts per gran abtessa

\numero{34} Ne les auran poetes per lur art

\numero{35} An les aquells metent vicis apart

\numero{36} Obrant virtut per amor de bonessa

\numero{37} Res no dubtant viciosa vergonya

\numero{38} Mas solament amant virtuos preu

\numero{39} E lome pech enaquest banch no seu

\numero{40} E qujnes luny lo gran delit se lonya

\end{estrofa}
