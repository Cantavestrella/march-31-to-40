\begin{estrofa}

\numero{1} Lome pel mon no muntan gran valer

\numero{2} Sens hauer bens bondat linatge gran

\numero{3} Mas la del mig ual mes quela restan

\numero{4} E no ual molt sens les altres hauer

\numero{5} Per ellas fan les dues molt prear

\numero{6} Car poder ual tant com es mjnjstrat

\numero{7} Ljmatge ual aytant com es honrat

\numero{8} La valor dom ho fa tot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no sera lom sabent de sonar

\numero{10} Sin algun tamps no sona esturment

\numero{11} Car per uoler sonar lo nom nos pren

\numero{12} Mas lesturment sonant be acordar

\numero{13} Tot en axi aquell quj dins si ual

\numero{14} Pobre de bens e dauillat limatge

\numero{15} No tels areus per mostrar gran coratge

\numero{16} Enla virtut ques nomena moral

\end{estrofa}



\begin{estrofa}

\numero{17} [32r] Son e seran molts dun altre cabal

\numero{18} Hauent\sout{s} molts bens e dalta sanc fauor

\numero{19} E ualent poch han la part no mellor

\numero{20} Car sens lom bo quant pot hauer es mal

\numero{21} E moltes ueus han la colpa natura

\numero{22} Car fara bo tal que valer no sab

\numero{23} Negun saber no pot viuren llur cap

\numero{24} Sens colpa llur de valler han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels estrems al mjg virtut atura

\numero{26} Molt greu dobrar yentre pochs conegut

\numero{27} Per ell saber nos hom per bo tengut

\numero{28} Mas fets hobrant forans dins tal mesura

\numero{29} Aytant es larch lom menys de fer larguesa

\numero{30} Com es escas sino fall en despendre

\numero{31} Vicis virtuts per actes san apendre

\numero{32} Apres lo fet es llur potenca presa

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conqueran virtuts per gran abtesa

\numero{34} Neles hauran poetes per llur art

\numero{35} Han les aquells metents vicis apart

\numero{36} Obrant uirtut per amor de bonesa

\numero{37} Res no duptant uiciosa uergonya

\numero{38} Mas solament amant uirtuos preu

\numero{39} E lome pech en aquest banch no seu

\numero{40} E qujn es luny lo gran delit se lonya

\end{estrofa}



\begin{estrofa}

\numero{41} [32v] Lir entre carts totstemps fare ma ponya

\numero{42} Que la dolor james de mjs partesqua

\numero{43} E no penseu que mon cas enferesqua

\numero{44} Car maior dan mereix ma gran vergonya.

\end{estrofa}
