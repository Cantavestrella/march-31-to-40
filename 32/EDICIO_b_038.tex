\begin{estrofa}

\numero{1} Pvjar no pot / algu en molt, valer

\numero{2} sens hauer, bens / bondat, linatge, gran

\numero{3} mas la del mig / val mes, que la restan

\numero{4} he no val molt / sens les altres, hauer

\numero{5} per, ellas fan, les dues, molt prear

\numero{6} car poder val / tant com es, ministrat

\numero{7} linatje val / tant com es mes, honrat

\numero{8} la valor d'hom / ho fa tot, graduar

\end{estrofa}



\begin{estrofa}

\numero{9} [30r] Mas no sera / l'hom, sabent, de sonar

\numero{10} si'nalgun temps / no sona, esturmen

\numero{11} car per voler / sonar, lo nom, nos pren

\numero{12} mas l'strument / sonant, be acordar

\numero{13} tot, en axi / aquell que dins si, val

\numero{14} pobre de bens / he d'auiltat, linatje

\numero{15} no tel's arreus \cite{ref29} / per mostrar, gran, coratje

\numero{16} en la virtut / quel hauer, d'hom, hi cal

\end{estrofa}



\begin{estrofa}

\numero{17} Son, he seran / molts d'un altre cabal

\numero{18} hauents molts bens / he d'alta, sanch, fauor

\numero{19} he valent poch / han la part, no millor

\numero{20} car, sens l'hom, bo / quant pot hauer, es mal

\numero{21} he moltas veus / han la colpa, natura

\numero{22} car, fora bo / tal, que valer, no sap

\numero{23} ningun saber / no pot viure'n lur cap

\numero{24} sens colpa lur / de valer, han fretura.\footnote{\textit{Tant aquest mot com
}natura\textit{ en el vers 21 apareixen amb la }a\textit{ final separada}: natur
a; fretur a.}

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels extrems / al mig, virtut, atura

\numero{26} molt greu d'obrar / y entre pochs, conegut

\numero{27} per ell saber / consegueix / hom, virtut

\numero{28} mas fets, obrant / forans, dins, tal mesura

\numero{29} ay tant es larch / hom, menys, de fer, larguesa

\numero{30} com es escas / sino fall, en despendre

\numero{31} vicis, virtuts / per actes, s'an, apendre

\numero{32} apres lo fet / es l'impoten\c{c}a, presa.

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conquerran / virtuts, per gran, abtesa

\numero{34} ne les hauran / poetes, per lur art

\numero{35} han les aquells / metent, vicis, apart

\numero{36} obrant, virtuts, per amor, de bonesa

\numero{37} [30v] mes no duptant / viciosa vergonya

\numero{38} mas soloment\footnote{\textit{Per} solament.} / amant, virtuos preu

\numero{39} hel'homepech / en aquest banch, no seu

\numero{40} he qui es luny / lo gran delit, s'alonya

\end{estrofa}
