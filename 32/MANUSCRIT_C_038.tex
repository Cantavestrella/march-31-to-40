\begin{estrofa}

\numero{1} Pujar no pot / algu en molt ualer

\numero{2} sens hauer bens / bondat linatge gran,

\numero{3} [47v] mas la del mig / ual mes quela restan

\numero{4} e no ual molt / sen les altres hauer

\numero{5} per ella's fan / les dues molt prear

\numero{6} car poder ual / tant com es ministrat

\numero{7} linatge ual tant / com es mes honrat

\numero{8} la ualor d'hom / ho fa tot graduar.

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no sera l'hom / saben de sonar

\numero{10} si n'algun temps / no sona esturmen

\numero{11} car per uoler / sonar lo nom nos pren

\numero{12} mas l'instrument / sonant be acordar

\numero{13} tot en axi / aquell de dins si ual

\numero{14} pobre debens / he d'auiltat linatje

\numero{15} no tel's arreus / per mostrar gran coratje

\numero{16} enla uirtut / quel hauer d'hom hi cal.

\end{estrofa}



\begin{estrofa}

\numero{17} Son e seran / molts d'un altre cabal

\numero{18} hauents molts bens / e d'altra sanch fauor

\numero{19} e ualent poch / han la part no millor

\numero{20} car sens l'hom bo / quant pot haueresmal

\numero{21} e moltas ueus / han la colpa natura

\numero{22} car fora bo / tal que ualor no sap

\numero{23} [48r] ningun saber / no pot uiure'n lur cap

\numero{24} sens colpa lur / de ualer han fretura.

\end{estrofa}



\begin{estrofa}

\numero{25} Entre l's extremps / al mig uirtut atura

\numero{26} molt greu d'obrar / y entre pochs conegut

\numero{27} per ell saber / consegueix hom uirtut

\numero{28} mas fets obrant / forans dins tal mesura

\numero{29} aytant es larch / hom menys de fer larguesa

\numero{30} com es escas / sino fall en despendre

\numero{31} uicis uirtuts / per actes s'an apendre

\numero{32} apres lo fet / es l'impoten\c{c}a presa.

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conquerran / uirtuts per gran abtesa

\numero{34} neles hauran / poetes per lur art

\numero{35} hanles aquells / metent uicis apart

\numero{36} obrant uirtuts / per amor de bonesa

\numero{37} mes no duptant / uiciosa uergonya

\numero{38} mas solament / amant uirtuos preu

\numero{39} e l'home pech / en aquest banch no seu

\numero{40} e qui es luny / lo gran delit s'alonya

\end{estrofa}
