\begin{estrofa}

\numero{1} Pujar no pot algu en molt valer

\numero{2} sens hauer bens bondat linajegran

\numero{3} mas ladel mig val mes que la restan

\numero{4} eno val molt sens les altres hauer

\numero{5} Per ellas fan les dues molt prear

\numero{6} car poder valtant com es ministrat

\numero{7} linajeval tant com es \interlin{mes} honrat

\numero{8} la valor dom ho fatot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no sera lom sabent de sonar

\numero{10} sinalgun temps no sona esturmen

\numero{11} car pervoler sonar lo nom nos pren

\numero{12} mas linstrument sonant be acordar

\numero{13} [29v] Tot enaxi aquell qui dins sival

\numero{14} pobre de bens e dauilat linaje

\numero{15} no tels arreus per mostrar gran coratje

\numero{16} en la virtut quel hauer dom hi cal

\end{estrofa}



\begin{estrofa}

\numero{17} Son e seran molts dun altre cabal

\numero{18} hauents molts bens edalta sanch fauor

\numero{19} evalent poch / hanlapart no millor

\numero{20} car sens lom bo quant \interlin{pot} hauer\footnote{hauer, \textit{corregit sobre una
paraula anterior raspada.}} es mal

\numero{21} Emoltas veu han la culpa natura

\numero{22} car forabo talquevaler no sab

\numero{23} ningun saber nopot viuren lur cap

\numero{24} sens culpalur devaler han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels extrems al mig virtut atura

\numero{26} molt greu dobrar y entre pochs conegut

\numero{27} per ell saber no consegueix hom virtut

\numero{28} mas fets obrant forans dins tal mesura

\numero{29} Aytant es larch hom menys de fer larguesa

\numero{30} com es escas sino fall endespendre

\numero{31} vicis virtuts peractes san apendre

\numero{32} apres lo fet es lur poten\c{c}a presa

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conquiren virtuts per gran abtesa

\numero{34} no les hauran poetes per lur art

\numero{35} hanles aquells metent vicis apart

\numero{36} obrant virtuts\footnote{virtuts, \textit{corregit sobre una paraula anterior
raspada.}} per amor de bonesa

\numero{37} Res no duptant viciosa vergonya

\numero{38} mas solament amant virtuospreu

\numero{39} [30r] el ompech / enaquestbanch no seu

\numero{40} equi es luny / lograndelit salonya

\end{estrofa}



\begin{estrofa}

\numero{41} Lir entre carts tostemps faremaponya

\numero{42} queladolor james demis partesca

\numero{43} eno penseu que moncas enferesca

\numero{44} car majordan mereix magran vergonya

\end{estrofa}
