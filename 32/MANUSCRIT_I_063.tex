\begin{estrofa}

\numero{1} PVjar\footnote{\textit{La} j\textit{ esmena una }g\textit{. Al marge}: pujar.}
no pot / algu en mes valer

\numero{2} Sens hauer bens / bondat\sout{s} linatge gran

\numero{3} Mas la del mig / val mes que lo restan

\numero{4} Eno val molt / sens les altres hauer

\numero{5} Per ella fan / les dues molt prear

\numero{6} Car poder val / tant com es minjstrat

\numero{7} Linatge val / aytant com es honrat

\numero{8} La valor dom / ho fa tot graduhar

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no sera / lom sabent de sonar

\numero{10} Sin algun temps / no sona jnstrumen

\numero{11} Car per voler / sonar lo nom nos pren

\numero{12} Mas be saber / lesturment ha cordar

\numero{13} Tot en axi / lome qui dins si val

\numero{14} Pobre de bens / e de molt alt linatge

\numero{15} No tels arre\textit{us} / per mostrar son coratge

\numero{16} En la virtut / que lauer dom hi cal

\end{estrofa}



\begin{estrofa}

\numero{17} [77r] Son e seran / molts dun altre cabal

\numero{18} Hauent molts bens / e dalta sanch fauor

\numero{19} E valen poch / en la part no mjllor

\numero{20} Car sens lom bo / quant pot hauer es mal

\numero{21} Mas alas ve\textit{us} / han la culpa natura

\numero{22} Car fora bo / tal que valer no sab

\numero{23} Degun saber / no pot viuren lur cap

\numero{24} Sens falta lur / de valer han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels estrems / virtut enmig atura

\numero{26} Molt greu dobrar / hentre pochs conegut

\numero{27} Per ell saber / nos hom p\textit{er} bo tengut

\numero{28} Mas fets obrant\sout{s} forans dins tal mesura

\numero{29} Aytant es larch lom / me\textit{n}ys de fer larguesa

\numero{30} Com es lauar / sino fall endespendre

\numero{31} Vicis virtuts / p\textit{er }actes san apendre

\numero{32} Per lacte fet / es lur potenca presa

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conquerran / virtuts per gran scienca

\numero{34} Ne les hauran / poetes per lur art

\numero{35} Han les aquells / matent vicis apart

\numero{36} Amants assi / menys presants la tamenca

\numero{37} Dels jugadors / reffrenants ab vergonya

\numero{38} Mas solament / per no guastar lur preu

\numero{39} Elome pech / en aquest banch no seu

\numero{40} Equin es luny / lo gran delit se lonya

\end{estrofa}
