\poema{(E 79) [32]}\\

\begin{rubrica}

Moss\textit{en} ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Puiar no pot / algu en molt valer\footnote{\textit{Es tracta de la poesia 32}
L'ome pel m\'{o}n no munta en gran valer.}

\numero{2} sens hauer bens / bondat linatge gran

\numero{3} mas la del mig / val mes que lo restan

\numero{4} e no val se poch / sens les altres hauer

\numero{5} per ellas fan / les dues molt prear

\numero{6} car lauer val / tant com es ministrat

\numero{7} linatge val / aytant com es honrrat

\numero{8} la valor dom / ho fa tot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} [115v] May no sera / lom sabent de sonar

\numero{10} sin algun temps / no sona jnsturmen\footnote{\textit{Escrit inicialment}
sturment.}

\numero{11} car per voler / sonar lo nom nos pren

\numero{12} menys de saber / lesturment acordar

\numero{13} Tot en axi / lome qui dins si val

\numero{14} pobre de bens / e de petit linatge

\numero{15} no tels arreus / per mostrar son coratge

\numero{16} enla virtut / quel auer dom hi cal

\end{estrofa}



\begin{estrofa}

\numero{17} Son e seran / molts dun altre cabal

\numero{18} hauents molts bens / e dalta sanch fauor

\numero{19} e valent poch / han la part no mellor

\numero{20} car sens lom bo / quant pot hauer es mal

\numero{21} E ales veus / hala colpa natura

\numero{22} car fora bo / tal que valer no sab

\numero{23} degun saber / no pot viuren llur cap

\numero{24} sens falta llur / de valer han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Esperiment / virtut en lom procura

\numero{26} cahent en lom / ques dispost a virtut

\numero{27} e sens aquest / tot hom porta escur

\numero{28} pera cobrir / lo seny o // loradura

\numero{29} Aytant es bo / lom sens sperien\c{c}a

\numero{30} com es dolent si pertal nos demostra

\numero{31} virtuts morals / no son menys dela mostra

\numero{32} lactes primer / que no es la poten\c{c}a

\end{estrofa}



\begin{estrofa}

\numero{33} [116r] Nos conquerran / virtuts per gran scien\c{c}a

\numero{34} Ne les hauran / poetes per llur art

\numero{35} han les aquells / metents vicis apart

\numero{36} amants a si / metents vicis apart\footnote{\textit{Aquest hemistiqui i els dos
versos seg\"{u}ents han estat reescrits}.}

\numero{37} Dels jutjadors / refrena\textit{n}ts ab vergonya

\numero{38} mas solament / per no gu\interlin{a}star lur preu

\numero{39} e lome pech / en aquest banch no seu

\numero{40} e qui nes lluny / lo gran delit se llonya

\end{estrofa}
