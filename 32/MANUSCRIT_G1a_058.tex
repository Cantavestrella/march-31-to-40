\begin{estrofa}

\numero{1} Lome pel mon no muntan gran valer

\numero{2} sens hauer bens bondat linatge gran

\numero{3} mas ladenmig val mes que lo restan

\numero{4} e no val molt sens los altres hauer

\numero{5} per ellas fan les dues molt prear

\numero{6} car poder val tant com es mjnistrat

\numero{7} linatge val aytant com es onrrat

\numero{8} la valor dom ho fa tot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} [90r] Mas no sera lom sabent de sonar

\numero{10} sin algun temps no sonasturment

\numero{11} car per voler sonar lo nom nos pren

\numero{12} mas lestrument sonant be acordar

\numero{13} tot en axi \interlin{com} lom que\footnote{\textit{Mot corregit damunt} quj.} dins si
val

\numero{14} pobre de bens ede vila linatge

\numero{15} no tels arreus per mostrar gran coratge

\numero{16} en la virtud ques nomena moral

\end{estrofa}



\begin{estrofa}

\numero{17} Son eseran molts dun altre cabal

\numero{18} hauent molts bens e dalta sanch fauor

\numero{19} e valent poch han la part no mjllor

\numero{20} car sens lom bo quant pot hauer es mal

\numero{21} e moltes veus ha la culpa natura

\numero{22} car fora bo tal que valer no sap

\numero{23} nengun saber no pot viuren llur cap

\numero{24} sens culpa llur de valer han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels estrems al mig virtud atura

\numero{26} molt greu dobrar y entre pochs conegut

\numero{27} per ell saber nos hom per bo tengut

\numero{28} mas fets obrant forans dins tal mesura

\numero{29} aytant es llarch lom menys defer larguesa

\numero{30} com es escas sino fall en despendre

\numero{31} vi\c{c}is virtuts per actes fan apendre

\numero{32} apres lo fet es lur poten\c{c}a presa

\end{estrofa}



\begin{estrofa}

\numero{33} [90v] Nos conquerran virtuts per granabteza

\numero{34} ne les hauran poetes per lur art

\numero{35} han les aquells metents vicis apart

\numero{36} obrant virtud per amor de bonesa

\numero{37} res no dubtant viciosa vergonya

\numero{38} mas solament amant virtuos preu

\numero{39} e lome pech en aquest banch no seu

\numero{40} e quin es luny lo gran deljt se lonya

\end{estrofa}
