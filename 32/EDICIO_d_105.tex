\poema{(d 105) [32]}\\

\begin{rubrica}

CANTO MORAL TERCERO.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} P\textsc{v}jar no pot / algu en molt valer

\numero{2} sens hauer bens / bondat,linatge,gran,

\numero{3} mas la de'n mig / val mes que la restan

\numero{4} e no val molt / sens les altres hauer.

\numero{5} Per ella'es fan / les dues molt prear

\numero{6} car poder val / tant com es ministrat

\numero{7} linatje val / ay tant com es honrrat

\numero{8} la valor d'hom / ho fa tot graduar.

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  Mas no sera / l'hom sabent de sonar

\numero{10} si en algun temps / no sona esturmen,

\numero{11} car per voler / sonar,lo nom nos pren

\numero{12} mas l'istrument / sonant be acordar.

\numero{13} Tot en axi / aquell que dins si val

\numero{14} pobre de bens / e de auiltat linatje

\numero{15} no te'ls arreus / per mostrar son coratje

\numero{16} en la virtut / quel hauer d'hom hi cal.

\end{estrofa}



\begin{estrofa}

\numero{17} [145r] \textparagraph\  Son e seran / molts d'un altre cabal

\numero{18} hauents molts bens / e d'altra sanch valor,

\numero{19} e valent poch / han la part no millor:

\numero{20} car sens l'hom bo / quant pot hauer es mal.

\numero{21} E moltes veus / ha la colpa natura

\numero{22} car fora bo / tal que valer no sab

\numero{23} ningun saber / no pot viure'n llur cap

\numero{24} sens colpa llur / de valer han fretura.

\end{estrofa}



\begin{estrofa}

\numero{25} \textparagraph\  Entrels extrems / al mig virtut atura

\numero{26} molt greu d'obrar / y entre pochs conegut

\numero{27} per ell saber / consegueix hom virtut

\numero{28} mas fets obrant / forans dins tal mesura.

\numero{29} Ay ta\textit{n}t es llarch / hom menys de fer llarguesa

\numero{30} com es escas / si no fall en despendre

\numero{31} vi\c{c}is,virtuts / per actes sean apendre

\numero{32} lacte's primer / apres poten\c{c}a'spresa.

\end{estrofa}



\begin{estrofa}

\numero{33} \textparagraph\  Nos conquerran / virtuts per gran abtesa

\numero{34} neles hauran / poetes per llur art,

\numero{35} han les aquells / metent vicis apart

\numero{36} obrant virtuts / per amor de bonesa

\numero{37} Mes no dubtant / vi\c{c}iosa vergonya

\numero{38} mas solament / amant virtuos preu

\numero{39} el home pech / en aquest banch no seu

\numero{40} e qui es lluny / lo gran delit se alonya.

\end{estrofa}
