\begin{estrofa}

\numero{1} Pujar nopot / algu en molt valer

\numero{2} sens auer bens / bondat llinyatge gran

\numero{3} mas lad\textit{e}l mig / val mes quelorestant

\numero{4} eno val molt / sens les altres auer

\numero{5} per ellas fan les dues molt prear

\numero{6} car poder val / tant com es ministrat

\numero{7} llinatge val / aytant com es honrrat

\numero{8} lavalor dom / o fa tot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no ss\textit{er}a lom / sabent sonar

\numero{10} sin alguntemps / no sona esturme\textit{n}ts

\numero{11} car per voler sonar / lo nom nos pren

\numero{12} mas de \tatxat{sturment / sonador} / saber lestrument
acordar

\numero{13} Tot enaxi / aquell que dins sival

\numero{14} pobre debens / edauilat llinatge

\numero{15} no tels arreu / per mostrar gran coratge

\numero{16} enlavirtut / que lauer dom y cap

\end{estrofa}



\begin{estrofa}

\numero{17} Son eseran / moltes dunaltre cabal

\numero{18} auents molts bens / edaltrasanch fauor

\numero{19} evalent poch / an lapart nomillor

\numero{20} car sen lombo / quant pot hauer es mal

\numero{21} E moltes veus dan / lacolpa natura

\numero{22} car fora bo tal / quevaler no sab

\numero{23} degun saber / no pot viure enllurcab

\numero{24} sens colpa llur / devaler anfretura

\end{estrofa}



\begin{estrofa}

\numero{25} [140r] Entrels estrems / almig virtut atura

\numero{26} molt greu dobrar / y entre pochs conegut

\numero{27} per ells saber / nos lom perbo tengut

\numero{28} mas fets obrant / forans dins tal mesura

\numero{29} Aytant es llarch / hom menys defer llarguesa

\numero{30} com es scas / sino fall endespendre

\numero{31} vincin\footnote{\textit{Per} vicis.} virtuts / per actes san apenrre

\numero{32} apres lo fet es llur / poten\c{c}a presa

\end{estrofa}



\begin{estrofa}

\numero{33} Nos conquerran / virtuts pergran obtesa

\numero{34} neles auran poetes / per lur art

\numero{35} sian les aquells metents vicis apart

\numero{36} obrant virtuts / peramor debonesa

\numero{37} Res no dubtant / viciosa vergonya

\numero{38} masolament / amant virtuos preu

\numero{39} elo me\tatxat{s} pech / enaquest banch no seu

\numero{40} equi nes lluny / logran delit sallonya

\end{estrofa}
