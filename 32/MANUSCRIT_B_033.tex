\begin{estrofa}

\numero{1} Lome pel mon no muntan gran valor

\numero{2} Sens hauer bens bondat linatie gran

\numero{3} Mas la del mjtx val mes que lo restan

\numero{4} Val e no ha sens les altres hauer

\numero{5} Per ellas fan los altres molt prear

\numero{6} Car poder val tant com es ministrat

\numero{7} [28v] Linatie val aytant com es honrrat

\numero{8} La valor dom ho fa tot graduar

\end{estrofa}



\begin{estrofa}

\numero{9} Mas no sera lom sabent de sonar

\numero{10} Si nalgun temps no sona sturmen

\numero{11} Car per voler sonar lo nom nos pren

\numero{12} Mas lesturment sonant ben acordar

\numero{13} Tot en axi aq\textit{ue}ll que dins si val

\numero{14} Pobre de bens e dauilat linatje

\numero{15} Notels mijans per mostrar gran coratje

\numero{16} En la virtut ques nomena moral

\end{estrofa}



\begin{estrofa}

\numero{17} Son e seran molts dun altre cabal

\numero{18} Hauents molts bens e dalta sanch fauor

\numero{19} E valent poch an la part no millor

\numero{20} Car sens lom bo quant pot hauer es mal

\numero{21} E moltas veus han la colpa natura

\numero{22} Car fora bo tal q\textit{ue} valer no sab

\numero{23} Algun saber no pot viuren lur cap

\numero{24} Sens colpa lur de valer han fretura

\end{estrofa}



\begin{estrofa}

\numero{25} Entrels strems al mjtx virtut atura

\numero{26} Molt greu dobrar jentre pochs conegut

\numero{27} Pel ell saber nos hom per bo tengut

\numero{28} Mas fets obrant forans dins tal mesura

\numero{29} Aytant es larch lom menys de fer larguesa

\numero{30} Com es escas si no fall en despendre

\numero{31} Vicis virtuts per actes poden pendre

\numero{32} Apres lo fet et lur poten\c{c}a presa

\end{estrofa}



\begin{estrofa}

\numero{33} No conquer hom virtuts per sol abtesa

\numero{34} Ne les hauran portes per la lur art

\numero{35} Han les aq\textit{ue}lls metents vi\c{c}is apart

\numero{36} Obrants virtuts per amor de bonesa

\numero{37} [29r] Res no dubtant vi\c{c}iosa vergo\textit{n}ya

\numero{38} Mes solament amant virtuos preu

\numero{39} El home pech en aquest banch no seu

\numero{40} E qui nes lo lony lo gran delit salonya

\end{estrofa}
