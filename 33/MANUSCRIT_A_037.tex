\begin{estrofa}

\numero{1} Ens lo desig / decosa deshonesta

\numero{2} Don ve dolor / atot enamorat

\numero{3} Visch dolorit / desigant ser amat

\numero{4} E par ho be / car no\textit{us} vull desonesta

\numero{5} So que jo am / devos es vostre seny

\numero{6} Elos estats / de vostra vida casta

\numero{7} Molt no deman / car mon desig no basta

\numero{8} Sino en\c{c}o / que honestat ateny

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment / en vos amar menpeny

\numero{10} E no lo cors / ab voler desonest

\numero{11} Tiram a vos / vn amigable gest

\numero{12} Ab sentir prim / qui desperta desdeny

\numero{13} Tant esta pres / lo meu enteniment

\numero{14} Per molta part / del vostre que li alta

\numero{15} Quem toll sentir / em fa la carn malalta

\numero{16} Dun tal dormjr / quim tol lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara / lo meu egual talent

\numero{18} Pus mon de part / que nos cansa nes farta

\numero{19} Car lespirit / tot lo finit aparta

\numero{20} No es encors / lo seu contentament

\numero{21} De vos deman / la voluntat guanyada

\numero{22} Cella qui es / en larma jnfinida

\numero{23} La part damor / que pot esser perida

\numero{24} En lo meu cor / no si trobas for\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch / creix la sua flamada

\numero{26} Quant li son dats / molts fusts per quels aflam

\numero{27} E la donchs creix / e mostra maior fam

\numero{28} Car pot sorbir / cosa quel sia dada

\numero{29} Ne pren amj / car ma voluntat creix

\numero{30} Per los desigs / presentats en ma pensa

\numero{31} E remoguts / serials fetoffen\c{c}a

\numero{32} Car altra part / ma voluntat nos peix

\end{estrofa}



\begin{estrofa}

\numero{33} [187v] Mon pensament / mostre quem entresteix

\numero{34} Quant entre gents / estich mut e pensiu

\numero{35} Ladonchs amor / peix sos fills en lo nju

\numero{36} Que dins mon cap / es lonch temps que nodreix

\numero{37} Cest es aquell / voler sens algun terme

\numero{38} Per\c{c}o amor / demj nos partira

\numero{39} Aquellamor / quen nostra carn esta

\numero{40} No met al cor / lo no cansable verme

\end{estrofa}
