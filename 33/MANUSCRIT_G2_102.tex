\begin{estrofa}

\numero{1} Sens lo desig / decosa desonesta

\numero{2} donve dolor / atot enamorat

\numero{3} visch dolorit / desijant ss\textit{er }amat

\numero{4} epar hobe / quenous vulldesonesta

\numero{5} Lo que pus am de vos es v\textit{ost}re seny

\numero{6} elos estats / dev\textit{os}tra vida casta

\numero{7} molt no deman / car mon desig no basta

\numero{8} sino en\c{c}o / que honestat atteny

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment / envos amar menpeny

\numero{10} eno lo cos / ab lo voler deshonest

\numero{11} tiram avos / vn amigable gest

\numero{12} ab sentir prim / qui desperta desdeny

\numero{13} Tant esta pres / lo meu entenime\textit{n}t

\numero{14} per moltapart / del v\textit{ost}re queli alta

\numero{15} quem tol sentir / emfa la carn malalta

\numero{16} duntal dormir / que pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara lo meu / egual talent

\numero{18} puix mon depart / que nos cansa nes farta

\numero{19} car lesperit / tot lo finit aparta

\numero{20} no es entot / lo seu contentament

\numero{21} Devos deman / lavoluntat guanyada

\numero{22} cella qui es / enlarma jnfinida

\numero{23} Ja part damor / quepot esser partida

\numero{24} en lo meu cor / noy es molt esfor\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} [141r] Sicom lo foch creix / la sua flamada

\numero{26} quant li son dats molts fusts perquels aflam\tatxat{a}

\numero{27} elladonchs crex e mostrmajor fam\tatxat{a}

\numero{28} com pot sorbir cosa quels sia dada

\numero{29} Ne pren ami carmavoluntat crex

\numero{30} per los desigs presentats en la pensa

\numero{31} e remoguts /serialofer offenssa

\numero{32} car daltra part / mavoluntat nos peix

\end{estrofa}



\begin{estrofa}

\numero{33} Monpensame\textit{n}t / mostraque mentrestexch

\numero{34} quant entre gents / estich mut epenssiu

\numero{35} lla donchs amor / peix sos fills en lo niu

\numero{36} quedins mon cap / llonch temps aquenodrex

\numero{37} cest es aquell voler / sens algun terme

\numero{38} per\c{c}o amor / enmi nosaparta

\numero{39} car lamador / quison voler fart ha

\numero{40} nol rou damor / lo nocansable verme

\end{estrofa}
