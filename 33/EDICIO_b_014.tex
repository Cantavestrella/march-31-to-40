\begin{estrofa}

\numero{1} S\textsc{e}ns lo desig / de cosa deshonesta

\numero{2} don ve dolor / a tot, enamorat

\numero{3} visch dolorit / desijant ser amat

\numero{4} he par ho be / que nous vull deshonesta

\numero{5} \c{c}o que mes am / de vos, es vostre seny

\numero{6} he los stats / de vostra vida casta

\numero{7} molt no deman / car mon desig no basta

\numero{8} sino en\c{c}o / que honestat, ateny

\end{estrofa}



\begin{estrofa}

\numero{9} [12r] Lenteniment / en vos amar me'npeny

\numero{10} he no lo cos / ab voler deshonest

\numero{11} tiram ha vos / vn amigable gest

\numero{12} ab sentir prim / qui desperta desdeny

\numero{13} tant esta pres / lo meu enteniment

\numero{14} per molta part / del vostre que li alta

\numero{15} quem tol sentir / hem fa la carn malalta

\numero{16} d'un tal morir / que pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara / lo meu hegual, talent

\numero{18} puix mou depart / que nos cansa nes farta

\numero{19} car, lesperit / tot lo finit, aparta

\numero{20} no es en tot / lo seu contentament

\numero{21} de vos deman / la voluntat guanyada

\numero{22} cella qui es / en larma, infinida

\numero{23} la part d'amor / que pot esser partida

\numero{24} en lo meu cor / noy es molt esfor\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch / creix la sua flamada

\numero{26} qua\textit{n}t li son dats / molts fusts per q\textit{ue}ls aflam

\numero{27} he lladonchs creix / ha mostra major fam

\numero{28} com pot surbir / cosa quel sie dada

\numero{29} ne pren ami / car ma valuntat creix

\numero{30} per los desigs / presentats en ma pensa

\numero{31} he, remoguts / serial fer offensa

\numero{32} car d'altra part / ma voluntat nos peix

\end{estrofa}
