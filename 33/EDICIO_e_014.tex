\poema{(e 14) [33]}\\

\begin{rubrica}

DE AMOR CANT. XIIII.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} S\textsc{e}ns lo desig / de cosa deshonesta

\numero{2} d'hon ve dolor / a tot enamorat

\numero{3} visch dolorit / desijant ser amat

\numero{4} e par ho be / que nous vull deshonesta.

\numero{5} \c{c}o que mes am / de vos es vostre seny

\numero{6} e los estats / de vostra vida casta

\numero{7} molt no deman / car mon desig no basta

\numero{8} sino en \c{c}o / que honestat ateny.

\end{estrofa}



\begin{estrofa}

\numero{9} L'enteniment / en vos amar m'empeny

\numero{10} e no lo cors / ab voler deshonest:

\numero{11} tiram'auos / vn amigable gest

\numero{12} ab sentir prim / qui desperta desdeny.

\numero{13} Tant esta pres / lo meu enteniment

\numero{14} per molta part / del vostre que li alta

\numero{15} quem tol sentir / hem fa la carn malalta

\numero{16} d'un tal dormir / que pert lo sentiment.

\end{estrofa}



\begin{estrofa}

\numero{17} [15r] No cessara / lo meu egual talent

\numero{18} puix mou depart / que nos cansa n'es farta

\numero{19} car l'esperit / tot lo finit aparta

\numero{20} no es en cos / lo seu contentament.

\numero{21} De vos deman / la voluntat guanyada

\numero{22} cella qui es / en l'arma infinida

\numero{23} la part d'Amor / que pot esser partida

\numero{24} en lo meu cor / noy es molt esfor\c{c}ada.

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch / creix la sua flamada

\numero{26} quant li son dats / molt fusts per quels aflam

\numero{27} ella donchs creix e mostra major fam

\numero{28} com pot sorbir / cosa que'l sia dada.

\numero{29} Ne pren a mi / car ma voluntat creix

\numero{30} per los desigs / presentats en ma pensa

\numero{31} e remoguts serials fer offensa:

\numero{32} car d'altra part / ma voluntat nos peix.

\end{estrofa}



\begin{estrofa}

\numero{33} Mon pensament / mostra quem'entristeix

\numero{34} quan entre gents / estich mut e pensiu,

\numero{35} la donchs Amor / peix sos fills en lo n\'{\i}u

\numero{36} que dins mon cap / ha lonch temps  que nodreix.

\numero{37} Est es aquell / voler sens algun terme

\numero{38} per \c{c}o qu'Amor / de mi james nos parta

\numero{39} car l'amador qui son voler no farta

\numero{40} nol rou d'amor / lo no cansable verme.

\end{estrofa}
