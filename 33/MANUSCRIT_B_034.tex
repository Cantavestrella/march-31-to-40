\begin{estrofa}

\numero{1} Sens lo desitx de cosa desonesta

\numero{2} Don ve dolor atot anamorat

\numero{3} Visch dolorit desijant ser amat

\numero{4} E par ho be car totaus vull honesta

\numero{5} So q\textit{ue} yo am de vos es vostre seny

\numero{6} E los stats de vostra vida casta

\numero{7} Molt no deman car mon desitx nobasta

\numero{8} Sino en so que honestat ateny

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment en vos amar menpeny

\numero{10} E no lo cos ab voler desonest

\numero{11} Tirant me vos vn amigable gest

\numero{12} Ab sentir prim qui desperta desdeny

\numero{13} Tant sta pres lo meu entenjment

\numero{14} Per molta part d\textit{e}l vostre que li alta

\numero{15} Quem tol sentir em fa la carn malalta

\numero{16} Dun tal dormjr q\textit{ue} pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara lo meu egual talent

\numero{18} Puys mon deport que nos cansa nes farta

\numero{19} Car lespirit tot lo finjt aparta

\numero{20} No es en cos lo seu contentament

\numero{21} De vos daman la voluntat guanyada

\numero{22} [29v] La qual sta en larma jnfinida

\numero{23} La part damor que pot esser perida

\numero{24} En lo meu cor no si troba for\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch crex la sua flamada

\numero{26} Quant li son dats molts fusts p\textit{er} quels afflam

\numero{27} Ella donchs crex e mostra maior fam

\numero{28} Com pot sorbir cosa quel sia dada

\numero{29} Ne pren ami car ma voluntat crex

\numero{30} Per los desitxs presentats en ma pe\textit{n}sa

\numero{31} E remoguts serials fet offensa

\numero{32} Car daltra part ma voluntat nos pex

\end{estrofa}



\begin{estrofa}

\numero{33} Mon penssament mostra quem entrestex

\numero{34} Quant entre gents stich mut e pensiu

\numero{35} Ladonchs amor pex sos fills enlo nju

\numero{36} Que dins mon cap es lonch temps q\textit{ue} nodrex

\numero{37} Aquest voler es qui no consent terme

\numero{38} Per \c{c}o amor de mj nos partira

\numero{39} La part damor quen nostra carn sta

\numero{40} No crial cor lo no cansable verme

\end{estrofa}
