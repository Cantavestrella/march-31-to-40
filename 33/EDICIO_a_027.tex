\begin{estrofa}

\numero{1} \textparagraph\  Sens lo desig / de cosa desonesta\footnote{\textit{Poesia
composta per fragments de dues poesies. Segons els edicions cr\'{\i}tiques
actuals, aquesta poesia est\`{a} formada per 32 versos de la poesia 33 i 40 de la
poesia 16}.}

\numero{2} Don ve dolor / a tot enamorat

\numero{3} Visch dolorit / desijant ser amat

\numero{4} E par ho be / puix nous vull desonesta

\numero{5} So que yo am / de vos es vostre seny

\numero{6} E los estats / de vostra vida casta

\numero{7} Molt no deman / car mon desig no basta

\numero{8} Sino en so / que honestat ateny

\end{estrofa}

\begin{estrofa}

\numero{9} \textparagraph\  No cessara / lo meu egual talent

\numero{10} Per mon depart / que nos cansa nes farta

\numero{11} Car lesperit / tot lo finit aparta

\numero{12} No es en cos / lo seu contentament

\numero{13} De vos deman / la voluntat guanyada

\numero{14} Cella qui es / en larma infinida

\numero{15} La part damor / que pot esser perida

\numero{16} En lo meu cor / no si troba esforsada

\end{estrofa}



\begin{estrofa}

\numero{17} [46r] \textparagraph\  Si com lo foch / creix la sua flamada

\numero{18} Qua\textit{n}t li son dats / molts llenys perq\textit{ue}ls aflam

\numero{19} Ella donchs crex / e mostra major fam

\numero{20} Com pot sorbir / cosa quel sia dada

\numero{21} Ne pren ami / car ma voluntat crex

\numero{22} Per los desigs / presentats enla pensa

\numero{23} E remoguts / serials feta ofensa

\numero{24} Car daltra part / ma voluntat nos pex

\end{estrofa}



\begin{estrofa}

\numero{25} Mon pensament / mostra que sentrestex

\numero{26} Quant entre gents / estich mut e pensiu

\numero{27} Lla donchs amor / pex sos fills enlo niu

\numero{28} Que dins mon cap / llonch temps ha q\textit{ue} nodrex

\numero{29} Cest es aquell / voler sens algun terme

\numero{30} Per hon amor / de mi nos partira

\numero{31} Que aquell amor / quen vostra carn esta

\numero{32} No met al cor / lo no cansable verme

\end{estrofa}

\begin{estrofa}

\numero{33} [46v] \textparagraph\  Junt es lo temps / que mon goig es
co\textit{m}plit\footnote{Els versos 33-72 constitueixen el poema 16 (tret de la
tornada) de l'edici\'{o} de Bohigas, per\`{o} ac\'{\i} apareixen a
continuaci\'{o} dels versos corresponents al poema 33 d'aquesta edici\'{o} (vv.
1-32), tot formant el cap\'{\i}tol 27}

\numero{34} Enlo esguart / de so que he desijat

\numero{35} Car vist he \c{c}o / don era desperat

\numero{36} E molt pus bell / que dins mi no fon dit

\numero{37} Mas de present / cay en dolor nollenta

\numero{38} Fort es he tant / quel cap me fa mudar

\numero{39} Torbat del tot / mos passos he acuytar

\numero{40} E trop remey / mas de mi no sabsenta

\end{estrofa}



\begin{estrofa}

\numero{41} \textparagraph\  Si com lo xorch / qui deu souint smenta

\numero{42} Que li don fills / assenats y adrets

\numero{43} E quant los veu / esser folls e contrets

\numero{44} Noua dolor / dauant ell se presenta,

\numero{45} Na pres ami / que tostemps desigi

\numero{46} Dona seruir / hon cabes tot mon alt

\numero{47} E quant la veig / mon cor se jau malalta\footnote{\textit{Per} malalt.},

\numero{48} Crehent de cert / que nos dolra de mi.

\end{estrofa}



\begin{estrofa}

\numero{49} [47r] De gran tristor / sobres dolor me ve

\numero{50} Quem cal fugir / qualseuol lloch escur

\numero{51} E de gran por / ma pensa a fet tal mur

\numero{52} Quels pensaments / damnosos li dete

\numero{53} E son aytals / que si dells nos defen

\numero{54} Ben enfortit / sa forsa mal defesa

\numero{55} Tots entraran / sients a taula mesa

\numero{56} Tremolar ja /sent mon enteniment.

\end{estrofa}



\begin{estrofa}

\numero{57} De ser vensut / o sobrat nos defen

\numero{58} Es lo tardar / quels pensaments son glots

\numero{59} Sens ferse lloch / entrar volrien tots

\numero{60} Dells nos pot fer / en semps tal passamen

\numero{61} E sil pus flach / pensament es primer

\numero{62} Lenteniment / for\c{c}ara de fugir

\numero{63} En llur debat / es mon ben auenir

\numero{64} Llur pau es deport / daquell mon jorn darrer

\end{estrofa}



\begin{estrofa}

\numero{65} [47v] \textparagraph\  Pijor que mort / es vida sens plaher

\numero{66} Mas nom acus / que fenexca ma vida

\numero{67} Car mentre visch / no pot esser perida

\numero{68} Una dolor / junt ab algun esper

\numero{69} Nose don ve / mas de part de natura

\numero{70} Que no consent / contra mort gran esfor\c{c}

\numero{71} E per a\c{c}o / contra mon cas mesfor\c{c}

\numero{72} Per no complir / la mia desuentura

\end{estrofa}
