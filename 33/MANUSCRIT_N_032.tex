\begin{estrofa}

\numero{1} \textparagraph\  Sens lo desig de cossa denosta

\numero{2} Don ue dolor atot enamorat

\numero{3} Visch dolorit desijant esser amat

\numero{4} E parho be car no\textit{us} vull desonosta

\numero{5} Co que yo ham de vos es vostre seny

\numero{6} E los estats de vostra vida casta

\numero{7} Molt no deman car mon desig nobasta

\numero{8} Sino enco que honestat ateny.

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment envos amar menpeyn

\numero{10} E no lo cors ab voler desonest

\numero{11} Tirem avos vn amigable gest

\numero{12} Ab sentir prim que desperta desdeny

\numero{13} Tant esta pres lomeu enteniment

\numero{14} Per molta part del vostre quelj alta

\numero{15} Quem tol sentir em fa lacarn malalta

\numero{16} Dun tal dormir que pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} [65] No sesara lo meu egual talent

\numero{18} Puys mou depart que nos causa nes farta

\numero{19} Car lesperit tot lo finit aparta

\numero{20} No es encors lo seu contentament

\numero{21} De vos deman la voluntat guanyada

\numero{22} Cella qujes enlarma infinida

\numero{23} Lapart damor que pot esser perida

\numero{24} Enlomeu cor nosi troba forcada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch crex lasua flamada

\numero{26} Quant lj son dats molts fusts per quels aflam

\numero{27} E lauors crex emostra major fam

\numero{28} Com pot sorbir cossa quel sia dada

\numero{29} Ne pren amj car ma voluntat crex

\numero{30} Per los desigs presentats ama pensa

\numero{31} E remoguts serialfeta hofensa

\numero{32} Car daltra part ma voluntat nos pex

\end{estrofa}



\begin{estrofa}

\numero{33} Mon pensament mostra que mentristex

\numero{34} Quant entre gents estich mut e pensiu

\numero{35} La donchs amor pex sos fills enlo nju

\numero{36} Que dins mon cap es lonch temps que nodreix

\numero{37} Cest es aquell voler sens algun terme

\numero{38} Perco amor demj nos partira

\numero{39} Aquella mor quen nostra carnesta

\numero{40} Nomet alcor lono cansable verme

\end{estrofa}
