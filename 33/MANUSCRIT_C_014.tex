\begin{estrofa}

\numero{1} Sens lo desigs / de cosa desonesta

\numero{2} don ue dolor / a tot enamorat

\numero{3} uisch dolorit / desijant ser amat

\numero{4} et parho be / que nous uull desonesta

\numero{5} \c{c}o que mes am / de uos e uostre seny

\numero{6} e los stats / de uostra uida casta

\numero{7} molt no deman /car mon designo basta

\numero{8} sino en\c{c}o / que honestat atteny

\end{estrofa}



\begin{estrofa}

\numero{9} Lentendiment enuos amar m'empeny

\numero{10} e no lo cos / ab dolor desonest

\numero{11} tiram a uos / un amigable gest

\numero{12} ab sentir prim / qui desperta desdeny

\numero{13} tant esta pres / lo meu enteniment

\numero{14} per molta part / del uostre queli \sout{f}alta

\numero{15} quemtol sentir / hem fa la carn malalta

\numero{16} d'un tal morir / que pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} [23r] No cessara / lo meu egual talent

\numero{18} puix mou de part / que nos cansa nes farta

\numero{19} car l'esperit / tot lo fenit aparta

\numero{20} no es en tot / lo seu contentament.

\numero{21} de uos deman / la uoluntad guanyada

\numero{22} cella qui es / es l'arma infinida:

\numero{23} la part d'amor / que pot esser partida

\numero{24} en lo meu cor / noy es molt esfor\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} Sicom lo foch / creix la sua flammada

\numero{26} quant li son dats / molts fusts perq\textit{ue}ls aflam

\numero{27} e li adonchs creix / e mostra maior fam

\numero{28} com pot surbir / cosa quel sie dada

\numero{29} ne pren a mi / car ma uoluntat creix

\numero{30} per los desigs / presentats en ma pensa

\numero{31} eremoguts / serial fer offensa

\numero{32} car d'altra part / ma uoluntat nos peix

\end{estrofa}



\begin{estrofa}

\numero{33} Nou pensament / mostra quem entristeix

\numero{34} quant entre gents / estich mut he pensiu

\numero{35} lla donchs amor / peix son fills enlo niu

\numero{36} qui dins mon cap / ha lonch temps que nodreix

\numero{37} [23v] est es aquell uoler / sens algun terme

\numero{38} per\c{c}o qu'amor / demi iames nos parta

\numero{39} car l'amador / que son uoler no farta

\numero{40} nol rou d'amor / lo no cansable uerme

\end{estrofa}
