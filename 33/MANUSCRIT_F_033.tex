\poema{(F 33) [33]}\\

\begin{rubrica}

Altra

\end{rubrica}
  
\end{estrofa}


\begin{estrofa}

\numero{1} Sens lo desig de cosa desonesta

\numero{2} Don ue dolor atot enamorat

\numero{3} Visch dolorit desijant ser amat

\numero{4} E par ho be que nous vull desonesta

\numero{5} Co que yo am de uos es uostre seny

\numero{6} E los estats de uostra uida casta

\numero{7} Molt no deman car mon desig no basta

\numero{8} Sino en co que honestat ateny.

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment auos amar menpeny

\numero{10} E no lo cors ab uoler deshonest

\numero{11} Tiram auos vn amigable gest

\numero{12} Ab sentir prim quj desperta desdeny

\numero{13} Tant esta pres lo meu entenjment

\numero{14} Per molta part del uostre queli alta

\numero{15} Quem toll sentir em fa la carn malalta

\numero{16} Dun tal dormjr que pert lo sentjment

\end{estrofa}



\begin{estrofa}

\numero{17} [33r] No cessara lo meu egual talent

\numero{18} Puys mou de part que nos canssa ns\footnote{\textit{Es fa dif\'{\i}cil de
llegir, la soluci\'{o} m\'{e}s plausible pot ser} nis.} farta

\numero{19} Car lesperit tot lo finit aparta

\numero{20} No es encors lo seu contentament

\numero{21} De uos deman la voluntat guanyada

\numero{22} Cella quj es en larma jnfinida

\numero{23} La part damor que pot esser perida

\numero{24} En lo meu cor nosi trobasforcada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch creix la sua flamada

\numero{26} Quant li son dats molts fusts per quels aflam

\numero{27} E la donchs creix e mostra maior fam

\numero{28} Com por sorbir cosa quel sia dada

\numero{29} Ne pren amj car ma voluntat creix

\numero{30} Per los desigs presentas en ma penssa

\numero{31} E remoguts serial fet offenssa

\numero{32} Car daltra part ma voluntat nos peix

\end{estrofa}



\begin{estrofa}

\numero{33} Mon pensament mostra que mentristeix

\numero{34} Quant entre gents estich mut e pensiu

\numero{35} La donchs amor peix sos fills en lo nju

\numero{36} Que dins mon cap es loch temps que nodreix

\numero{37} Cest \interlin{es} aquell uoler sens negun terme

\numero{38} Per co amor de mj nos partira

\numero{39} Aquellamor quan nostra carn esta

\numero{40} No met al cor lo no cansable uerme

\end{estrofa}



\begin{estrofa}

\numero{41} [33v] Plena de seny mon entenjment ferme

\numero{42} En co que may amador lo ferma

\numero{43} E si fallesch veramor fallira

\numero{44} Mellor de tots haure nom sim conferme

\end{estrofa}
