\begin{estrofa}

\numero{1} s Ens lo desitx de cosa desonesta

\numero{2} Don ve dolor atot enamorat

\numero{3} Visch dolorit desijant ser amat

\numero{4} E par ho be car nous vull desonesta

\numero{5} So q\textit{ue} jo am de vos es vostre seny

\numero{6} E los stats de vostra vida casta

\numero{7} Molt no deman car mo\textit{n} desitx no basta

\numero{8} Sino en so q\textit{ue }desonestat ateny

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment en uos amar manpeny

\numero{10} Eno lo cos ab voler desonest

\numero{11} [33r] Tiram a vos vn amigable gest

\numero{12} Ab sentir prim qui desperta desdeny

\numero{13} Tant sta pres lo meu entenjment

\numero{14} Per molta part del vostre q\textit{ue} li alta

\numero{15} Quem tol sentir em fa la carn malalta

\numero{16} Dun tal dormir q\textit{ue} pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara lo meu egual talent

\numero{18} Puys mou d\textit{e} part q\textit{ue} nos can\c{c}a nes farta

\numero{19} Car lesperit tot lo finjt aparta

\numero{20} No es en cos lo seu contentament

\numero{21} De uos dema\textit{n} la voluntat gua\textit{n}yada

\numero{22} Cella qui es en larma jnfinjda

\numero{23} La part damor q\textit{ue} pot esser perida

\numero{24} En lo meu cor no si troba ffor\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch crex la sua flamada

\numero{26} Quan li son dats molts fusts p\textit{er} q\textit{ue}ls affla\textit{m}

\numero{27} E ladonchs crex e mostra maior fam

\numero{28} Car pot sorbir cosa quel sia dada

\numero{29} Ne pren amj car ma voluntat crex

\numero{30} Per los desitxs presentats en ma pensa

\numero{31} E remoguts serial fet offen\c{c}a

\numero{32} Car daltre past ma volu\textit{n}tat nos pex

\end{estrofa}



\begin{estrofa}

\numero{33} Mon pensament mostra qu\interlin{e}mentristex

\numero{34} Quant entre gents stich mut e pensiu

\numero{35} Ladonchs amor pex sos fills en lo nju

\numero{36} [33v] Que dins mon cap es lonch temps q\textit{ue }nodrex

\numero{37} Cast es aquell voler sens algu\textit{n} terme

\numero{38} Per\c{c}o amor de mi nos partira

\numero{39} Aquellamor quen nostra carn esta

\numero{40} No met al cor lo no cansable verme

\end{estrofa}
