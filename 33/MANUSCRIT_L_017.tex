\end{rubrica}

\begin{estrofa}

\numero{1} Sens lo desig / de cosa deshonesta

\numero{2} Don ve dolor / atot enamorat

\numero{3} Visch dolorit / desigant ser amat

\numero{4} E par ho be / car no\textit{us} vull deshonesta

\numero{5} \c{C}o que jo am / de vos es vostre seny

\numero{6} E los estats / de vostra vida casta

\numero{7} Molt no deman / car mon desig no basta

\numero{8} Sino en \c{c}o / que honestat ateny

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment / en vos amar menpeny

\numero{10} E no lo cors / ab voler deshonest

\numero{11} Tiram avos / vn amigable gest

\numero{12} Ab sentir prim / quj desperta desdeny

\numero{13} [148v] Tant esta pres / lo meu entenjment

\numero{14} Per molta part / del vostre que lj alta

\numero{15} Quem toll sentir / em fa la carn malalta

\numero{16} Dun tal dormir / quim tol lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara / lo meu egual talent

\numero{18} Puys mou de part / que nos cansa nes farta

\numero{19} Car lespirit / tot lo finit aparta

\numero{20} No es en cors / lo seu contentament

\numero{21} Deuos deman / la voluntat guanyada

\numero{22} Cella quj es / en larma jnfinjda

\numero{23} La part damor / que pot esser perida

\numero{24} En lo meu cor / no si troba sfor\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} Si com lo foch / creix la sua flamada

\numero{26} Quant lj son dats / molts fusts per quels afflam

\numero{27} E la donchs creix / emostra maior fam

\numero{28} Car pot sorbir / cosa quel sia dada

\numero{29} Ne pren amj / car ma voluntat creix

\numero{30} Per los desigs / presentats en ma pensa

\numero{31} E remoguts / serial fet offensa

\numero{32} Car daltra part / ma voluntat nos peix

\end{estrofa}



\begin{estrofa}

\numero{33} [149r] Mon pensament / mostra quem entristeix

\numero{34} Quant entre gents / estich mut e pensiu

\numero{35} La donchs amor / peix sos fills en lo niu

\numero{36} Que dins mon cap / es lonch temps que nodreix

\numero{37} Cest es aquell / voler sens algun terme

\numero{38} Per \c{c}o amor / de mj nos partira

\numero{39} Aquellamor / quen nostra carn esta

\numero{40} No met al cor / lo no cansable verme

\end{estrofa}
