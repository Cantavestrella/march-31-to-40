\begin{estrofa}

\numero{1} Sens lo desig / de cosa desonesta

\numero{2} don ve dolor / atot enamorat

\numero{3} visch dolorit / desijant ser amat

\numero{4} e par ho be / que nous vull desonesta

\numero{5} \c{C}o que pus am / de vos es vostre seny

\numero{6} e los estats / de vostra vida casta

\numero{7} molt no deman / car mon desig no basta

\numero{8} si no en \c{c}o / que honestat atteny

\end{estrofa}



\begin{estrofa}

\numero{9} Lenteniment / a vos amar menpeny

\numero{10} e no lo cors / ab voler desonest

\numero{11} tiram a vos / vn amigable gest

\numero{12} ab sentir prim / que desperta desdeny

\numero{13} Tant esta pres / lo meu enteniment

\numero{14} per molta part / del vostre quel asalta

\numero{15} quem tol sentir / hem fa la carn malalta

\numero{16} dun tal dormir / que pert lo sentiment

\end{estrofa}



\begin{estrofa}

\numero{17} No cessara / lo meu egual talent

\numero{18} puis mou de part / que nos cansa ne farta

\numero{19} car lesperit / \c{c}o ques finit aparta

\numero{20} e pren de nou / lo nouell pensament

\numero{21} De vos deman / la voluntat guanyada

\numero{22} \c{c}ella qui es / en larma jnfinida

\numero{23} la part damor / que pot esser perida

\numero{24} en lo meu cor / noyes molt esfor\c{c}ada

\end{estrofa}



\begin{estrofa}

\numero{25} [161r] Si com lo foch / creix la sua flamada

\numero{26} quant li son dats / molts fusts per quells aflam

\numero{27} e lladonchs creix / e mostra maior fam

\numero{28} com pot sorbir / cosa quel sia dada

\numero{29} Ne pren ami / car ma voluntat creix

\numero{30} per los desigs / presentats en la pensa

\numero{31} e remoguts / serials fer offensa

\numero{32} car daltra part / ma voluntat nos peix

\end{estrofa}



\begin{estrofa}

\numero{33} Mon pensament / mostra quem entresteix

\numero{34} quant entre gens / estich mut e pensiu

\numero{35} lladonchs amor / peix sos fills en lo niu

\numero{36} que dins mon cap / llonch temps ha que nodreix

\numero{37} Cest es aquell / voler sens algun terme

\numero{38} per \c{c}o amor / en mi no finara

\numero{39} car lamador / qui son voler farta

\numero{40} nol rou damor / lo no cansable verme

\end{estrofa}
