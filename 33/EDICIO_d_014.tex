\pagina{[20v]} DE AMOR CANTO . XIIII.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} S\textsc{e}ns lo desig / de cosa des honesta

\numero{2} D'honue dolor / atot enamorat

\numero{3} visch dolorit / desijant ser amat

\numero{4} e par ho be / que no vs vull deshonesta.

\numero{5} \c{C}o que pus am / de vos:es vostre seny

\numero{6} e los estats / de vostra vida casta

\numero{7} molt no deman / car mon desig no basta

\numero{8} sino en \c{c}o / que honestat ateny.

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  L'enteniment / en vos amar me'npeny

\numero{10} e no lo cos / ab voler deshonest:

\numero{11} tiram'auos / vn amigable gest

\numero{12} ab sentir prim / qui desperta desdeny.

\numero{13} Tant esta pres / lo meu enteniment

\numero{14} per molta part / del vostre que li alta

\numero{15} quem tol sentir / hem'fa la carn malalta

\numero{16} d'un tal dormir / que pert lo sentiment.

\end{estrofa}



\begin{estrofa}

\numero{17} \textparagraph\  No \c{c}essara / lo meu egual talent

\numero{18} puix mou depart / que nos cansa ne's farta

\numero{19} car le'spirit / tot lo finit aparta

\numero{20} no es en cos / lo seu contentament.

\numero{21} De vos deman / la voluntat guanyada

\numero{22} \c{c}ella qui es / en l'arma infinida

\numero{23} la part de Amor / que pot esser partida

\numero{24} en lo meu cor / noy es molt esfor\c{c}ada.

\end{estrofa}



\begin{estrofa}

\numero{25} [21r] \textparagraph\  Si com lo foch / creix la sua flamada

\numero{26} qua\textit{n} li son dats / molts fusts per quels aflam

\numero{27} ella donchs creix / e mostra major fam

\numero{28} com pot sorbir / cosa quel sia dada.

\numero{29} Ne pren ami / car ma voluntat creix

\numero{30} per los desigs / persentats\footnote{\textit{Per} presentats.} en ma pensa

\numero{31} e remoguts serials fer ofensa:

\numero{32} car d'altra part / ma voluntat nos peix.

\end{estrofa}



\begin{estrofa}

\numero{33} Mon pe\textit{n}sament / mostra quem'entristeix

\numero{34} quan entre gents / estich mut e pensiu

\numero{35} lla donchs Amor / peix sos fills en lo niu

\numero{36} q\textit{ue} dins mo\textit{n} cap / ha llonch te\textit{m}ps q\textit{ue}
nodreix.

\numero{37} Est'es aquell / voler sens algun terme

\numero{38} per\c{c}o Amor / de mi nos, partira

\numero{39} aquell Amor / que'n nostra carn esta

\numero{40} no met al cor / lo no cansable verme.

\end{estrofa}

\numero{43} e si fallesch / ver Amor fallir\'{a}

\numero{44} millor de tots / haure nom sim conferme.
