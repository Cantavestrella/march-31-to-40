\poema{(E 55) [31]}\\

\begin{rubrica}

Mossen ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Molt homens oig / clamarse de fortuna

\numero{2} e malahir / aquella per sos actes

\numero{3} volents ab ley / fer amigables pactes

\numero{4} tals quen lur be / en p\textit{er} tots temps fos vna

\numero{5} Volent mudar / sa propia natura

\numero{6} ques lalt baixar / y el baix muntar alt

\numero{7} e qui day\c{c}o / se dona gran desalt

\numero{8} leixe los bens / portants ab si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} Adeu no plach / hauer del mon tal cura

\numero{10} que noy jaquis / de terribles affanys

\numero{11} perdues son / en nombre mes quels guanys

\numero{12} nesguart del ver / lo seny es horadura

\numero{13} E lom girat / almundanal saber

\numero{14} ydoles creu / donant hi plena fe

\numero{15} yen bens mundans / creu esser complit be

\numero{16} volents\footnote{\textit{Grafia} s \textit{afegida per una altra m\`{a}
posteriorment}.} morir / solament per hauer

\end{estrofa}



\begin{estrofa}

\numero{17} [85v] Salgu posques / de fortuna tener

\numero{18} ab hun fort clau / sa roda quant es sus

\numero{19} fora li bo / anar amunt e jus

\numero{20} sofferint mals / per trobar tal mester

\numero{21} Mas los metalls / no han tanta vigor

\numero{22} que tan gran pes / ab llur for\c{c}a retinguen

\numero{23} ne ginys ne arts / no faran que no vinguen

\numero{24} \c{c}ells qui seuran / en la falsa honor

\end{estrofa}



\begin{estrofa}

\numero{25} E per null temps / perdra fortunardor

\numero{26} de fer llochs plans / deles terres e munts

\numero{27} e quant del tot / son desijat es junts

\numero{28} e no pot fer / pus ab sa gran furor

\numero{29} Torna reffer / desabitada casa

\numero{30} quen poch spay / hauia feta buyda

\numero{31} e lom desert / quen requir james cuyda

\numero{32} vmple graners / olis e vins embasa

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat / no deu portar spasa

\numero{34} com a foll hom / deu franch arbitre perdre

\numero{35} qui no creu \c{c}ert / que pot fortuna sperdre

\numero{36} los bens mouents / e los sehents arrasa

\numero{37} Mas lom del mon / per fortuna mirat

\numero{38} ha lo esguart / de riallosa cara

\numero{39} la part del plor / no ha vista encara

\numero{40} creu que sos ginys / lan de bens prosperat

\end{estrofa}
