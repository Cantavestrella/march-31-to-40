\begin{estrofa}

\numero{1} Molts homens hoig / clamarse de fortuna

\numero{2} e malehir aquella / per sos actes

\numero{3} uolents ab leys / fer amigable pactes

\numero{4} dona\textit{n}ts los be / enque tostemps fos una

\numero{5} no recordant / sa propria natura

\numero{6} qu'es l'alt baixar / e lo baix muntar alt

\numero{7} e qui d'a\c{c}o / se dona gran desalt

\numero{8} leixe los bens / portants ab si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} A deu no plach / hauer del mon tal cura

\numero{10} que noy iaquis / de terribles affanys

\numero{11} [50v] per dues son / en nombre mes quels guanys

\numero{12} en sguart del uer / lo seny es oradura

\numero{13} el hom girat/ al mon d'auol saber

\numero{14} ydoles creu / donantli plena fe

\numero{15} en bens mundans / creu esser complit be

\numero{16} uolent morir / solament te d'hauer.

\end{estrofa}



\begin{estrofa}

\numero{17} S'algu pogues / de fortuna tener

\numero{18} ab un fort clau / sa rota quant es sus

\numero{19} fora li bo / anar amunt e ius

\numero{20} sofferint mal / per trobar tal mester

\numero{21} mas los no tals / no han tanta uigor

\numero{22} que tant gran pes / ablur for\c{c}a retinguen

\numero{23} ne gynis nos fan / que d'alt e baix no uinguen

\numero{24} los qui seuran / en la falsa honor.

\end{estrofa}



\begin{estrofa}

\numero{25} Per nengun temps / perdra fortun'ardor

\numero{26} de fer lochs plans / deles timpes e munt

\numero{27} e quant del tot / son desijat es junt

\numero{28} e no pot fer / pus absa gran furor,

\numero{29} [51r] torna refer / deshabitada casa

\numero{30} qu'en poch spay / hauie\footnote{\textit{Ms. }hanie\textit{. Confusi\'{o}
habitual n/u.}} feta buyda

\numero{31} el hom desert / quen' riquir james cuyda

\numero{32} un ple graners / olis e uins embasa.

\end{estrofa}



\begin{estrofa}

\numero{33} E son costat / no deu portar spasa

\numero{34} com a foll hom / deu franch arbitre prendre

\numero{35} qui no creu cert / que fortuna pot pendre

\numero{36} los bens mouent / y los sehens arasa

\numero{37} mas l'hom del mon per fortuna mirat

\numero{38} ablo sguart / de riallosa cara

\numero{39} la part del plor / no ha uista encara

\numero{40} creu que sos ginys / l'han debe prosperat.

\end{estrofa}
