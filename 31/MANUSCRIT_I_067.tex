\poema{(I 67) [31]}\\

\begin{rubrica}

Mossen Auzias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Olt home veig / clamar de sa fortuna

\numero{2} E malahir / aquella per sos actes

\numero{3} Volents ab leys / fer amjgables pactes

\numero{4} Donant lurs be / e que tostemps fos vna

\numero{5} No recordant / sapropia natura

\numero{6} Ques lalt baxar / e lo baix muntar alt

\numero{7} Equi da\c{c}o / se dona gran desalt

\numero{8} Lexa los bens / portants en si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} [79v] Adeu no plach / del mon hauer tal cura

\numero{10} Que noy jaquis / de terribles affanys

\numero{11} Per dues son / en nombre mes q\textit{ue}ls guanys

\numero{12} Enuers lauer / lo seny es oradura

\numero{13} Elom girat / al mundanal saber

\numero{14} Jdoles creu / donant li plena fe

\numero{15} Quen bens mundans / cuydesser complit be

\numero{16} E vol morir / solament per hauer

\end{estrofa}



\begin{estrofa}

\numero{17} Silguns pogues / de fortuna tener

\numero{18} Ab vn fort clau / la roda com es sus

\numero{19} Ffora menys mal / anar amu\textit{n}t e jus

\numero{20} Sofferint mals / per cobrar tal mester

\numero{21} Mas los malalts / no han tanta vigor

\numero{22} Que tant gran pes / ab lur for\c{c}a retinguen

\numero{23} Ne ginys ne arts / no faran que no \interlin{a}vinguen

\numero{24} Los quey seran / en la falsa honor

\end{estrofa}



\begin{estrofa}

\numero{25} E per null temps / perdra fortunador

\numero{26} Deffer lonchs plants / deles terras amunt

\numero{27} E quant del tot / son desigat \sout{son} \interlin{es} junt

\numero{28} E no pot fer / pus ab sa gran furor

\numero{29} Torna reffer / desabitada cassa

\numero{30} Quen poch espay / ell laura feta buyda

\numero{31} Elom de sert / que enriquir no cuyda

\numero{32} Vmple greners / olis e vins enbassa\footnote{\textit{Al marge d'una altra
m\`{a}}: olins evins embasa.}

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat / no deu portar espassa

\numero{34} Com afoll hom / deu franch arbitre perdre

\numero{35} Que no creu cert / que fortuna pot perdre

\numero{36} Los bens mouents / e los scients embassa

\numero{37} Mas lom qui es / per fortuna mirat

\numero{38} Ab lo esguard / de riallosa cara

\numero{39} E daltra part / no aiuda encara

\numero{40} Creu que tots ginys / lan de bens prosperat

\end{estrofa}
