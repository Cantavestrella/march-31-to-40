\poema{(D 31) [31]}\\

\begin{rubrica}

C

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Molts homens oig clamarse de fortuna

\numero{2} emala hir / aquella per sos actes

\numero{3} volents ab leys / fer amigables pactes

\numero{4} donant los be / eque tostems fos una

\numero{5} [28v] No recordant sa propria natura

\numero{6} ques lalt baxar / e lo baixmuntar alt

\numero{7} equida\c{c}o sedonagran desalt

\numero{8} lexe los bens portants ab si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} A deu no plach hauer del montal cura

\numero{10} que noy jaquis de terribles affanys

\numero{11} perdues son en nombre mes quels guanys

\numero{12} nesguart delver lo seny es oradura

\numero{13} El homgirat al mundanal saber

\numero{14} ydoles creu donant hi plena fe

\numero{15} en bens mundans creu esser complit be

\numero{16} volent morir solament per hauer

\end{estrofa}



\begin{estrofa}

\numero{17} Sialgupogues de fortuna tener

\numero{18} ab vn fort clau saroda quant es sus

\numero{19} foralibo anar / amunt ejus

\numero{20} soferint mals pertrobar tal mester

\numero{21} Mas los metalls no han tantavigor

\numero{22} que tangranpes ab lur for\c{c}a retinguen

\numero{23} neginys no fan que dalt en baix no vinguen

\numero{24} los qui seuran en la falsa honor

\end{estrofa}



\begin{estrofa}

\numero{25} Per negun temps perdra fortunardor

\numero{26} de fer lochs plans de les timpes e munt

\numero{27} equant del tot sondesijat es junt

\numero{28} eno pot fer pus ab sagran furor

\numero{29} Torna refer / desabitada casa

\numero{30} quen poch espay hauia feta buyda

\numero{31} e lom desert quen requir james cuyda

\numero{32} umple graners / olis evins embasa

\end{estrofa}



\begin{estrofa}

\numero{33} [29r] En son costat no deu portar espasa

\numero{34} com a foll hom deu franch arbitre pendre

\numero{35} qui no creu cert que fortunapot perdre

\numero{36} los bens mouents y los sehents arrasa

\numero{37} Mas lomdel mon perfortuna mirat

\numero{38} ab lo esguart de riallosa cara

\numero{39} la part del plor no ha vista encara

\numero{40} creu que sos ginys lan de be prosperat

\end{estrofa}



\begin{estrofa}

\numero{41} Lir entre carts propri so comparat

\numero{42} a Tantalus percontinuu desig

\numero{43} no se perque tots dies hi affig

\numero{44} puys que mes prop compte desesperat

\end{estrofa}
