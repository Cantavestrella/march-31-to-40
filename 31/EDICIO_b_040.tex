\begin{estrofa}

\numero{1} Molts home\textit{n}s hoig / clamor\footnote{\textit{Per }clamar.} se, de
fortu\textit{n}a

\numero{2} he malehir / aquella, per sos actes

\numero{3} volenst\footnote{\textit{Per }volents.} ab leys \cite{ref32} / fer amigables,
pactes

\numero{4} donant los be / e que tostemps, fos vna

\numero{5} no recordant / sa propia, natura

\numero{6} que's lalt, baixar / he lo baix, muntar, alt

\numero{7} he qui da\c{c}o / se dona gan desalt,

\numero{8} leixe los bens / portants absi, fretura

\end{estrofa}



\begin{estrofa}

\numero{9} [32r] A deu no plach / hauer del mon tal cura

\numero{10} que noy jaquis / de terribles, affanys

\numero{11} perdues, son / en nombre, mes quels guanys

\numero{12} en'sguart del ver / lo seny es oradura

\numero{13} el hom girat / al mon d'auol saber

\numero{14} Ydoles, creu / donant, y plena fe

\numero{15} en bens mondans / creu esser, complit be

\numero{16} volent morir / solament, te d'hauer

\end{estrofa}



\begin{estrofa}

\numero{17} Si algu pogues / de fortuna, tener

\numero{18} ab vn fort clau / sa roda, quant, es sus

\numero{19} fora li bo / anar amunt, he jus

\numero{20} soferint mal / per trobar, tal mester

\numero{21} mas los no, tals / no han tanta, vigor

\numero{22} que tan gran pes / ab lur for\c{c}a retingen

\numero{23} ne'gynis\footnote{\textit{Per }ginys.} no fan / q\textit{ue} d'alt e\textit{n}
baix, no vingen

\numero{24} los qui seuran / en la falsa, honor

\end{estrofa}



\begin{estrofa}

\numero{25} Per negun temps / perdra fortuna'rdor

\numero{26} de fer lochs plans / deles timpes, he munt,

\numero{27} he quant del tot / son desijat, es junt

\numero{28} he no pot fer / pus ab sa gran, furor

\numero{29} torna refer / deshabitada, casa

\numero{30} que'n poch spay / havia feta, buyda

\numero{31} he lom desert / que'nriquir james cuyda

\numero{32} Vmple graners / olis, he vins, embasa \cite{ref33}

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat / no deu portar, spasa

\numero{34} com afoll hom / deu franch, arbitre pendre

\numero{35} qui no creu cert / que fortuna, pot pe\textit{n}dre

\numero{36} los bens mouents / y los sehents, arrasa

\numero{37} [32v] mas lom del mon / per fortuna, mirat

\numero{38} ab lo, sguart / de, riallosa, cara

\numero{39} la part del plor / no ha vista, encara

\numero{40} creu que sos ginys / lan de be, prosperat

\end{estrofa}
