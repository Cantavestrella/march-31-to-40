\poema{(d 106) [31]}\\

\begin{rubrica}

CANTO MORAL. IIII.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} M\textsc{o}lts home\textit{n}s hoyg / clamar se d\textit{e} fortuna

\numero{2} e maleir / aquella per sos actes,

\numero{3} volents ab leys / fer amigables pactes

\numero{4} donant los be / e que tostemps fos vna.

\numero{5} No recordant / sa propria natura

\numero{6} que's lo alt baixar / e lo baix muntar alt,

\numero{7} e qui d'a\c{c}o / se dona gran desalt

\numero{8} leixe los bens / portants abvsi fretura.

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  A Deu no plach / hauer del mon tal cura

\numero{10} que noy jaquis / de terribles affanys

\numero{11} perdues son / en nombre mes quels guanys

\numero{12} en'sguart del ver / lo seny es oradura.

\numero{13} El hom girat / al mondanal saber

\numero{14} ydoles creu / donant y plena fer

\numero{15} enbens mundans / creu esser complit be

\numero{16} volent morir / solament pels hauer.

\end{estrofa}



\begin{estrofa}

\numero{17} [146r] \textparagraph\  Si algu pogues / de fortuna tener

\numero{18} ab vn fort clau / sa roda quant es sus

\numero{19} fora li bo / anar amunt ejus

\numero{20} soferint mal / per trobar tal mester.

\numero{21} Mas los metals / no han tanta vigor

\numero{22} que tant gran pes / ab llur for\c{c}a retinguen

\numero{23} ne'nginys nos fan / q\textit{ue} d'alt,e baix,no vinguen

\numero{24} los qui seuran / en la falsa honor.

\end{estrofa}



\begin{estrofa}

\numero{25} \textparagraph\  Per nengun temps / perdra fortuna'rdor

\numero{26} de fer llochs plans / deles timpes e munt,

\numero{27} e quant del tot / son desijat es junt

\numero{28} e no pot fer / pus ab sa gran furor,

\numero{29} Torna refer / deshabitada casa

\numero{30} quen poch espay / hauie feta buyda

\numero{31} el hom disert,que'nriquir james cuyda

\numero{32} vmple graners / olis,e vins,embasa.

\end{estrofa}



\begin{estrofa}

\numero{33} \textparagraph\  En son costat / no deu portar espasa

\numero{34} com a foll hom / deu franch arbitre perdre,

\numero{35} qui no creu \c{c}ert / que fortuna pot pendre

\numero{36} los bens mouents / y los sehents arrasa.

\numero{37} Mas l'hom del mon / per fortuna mirat

\numero{38} ab lo esguart / de rriallosa cara

\numero{39} la part del plor / no ha vista encara

\numero{40} creu que sos ginys / l'han debe prosperat.

\end{estrofa}
