\begin{estrofa}

\numero{1} M\textsc{o}lts homens hoig / clamar se de fortu\textit{n}a

\numero{2} e malehir / aquella per sos actes.

\numero{3} volents ab leys / fer amigables pactes

\numero{4} donant los be / en que tostemps fos vna.

\numero{5} no recordant / sa propria natura

\numero{6} qu's l'alt baixar / e lo baix muntar alt,

\numero{7} e qui d'a\c{c}o / sedona gan\footnote{\textit{Per} gran.} desalt

\numero{8} leix e los bens / portants absi fretura.

\end{estrofa}



\begin{estrofa}

\numero{9} A Deu no plach / hauer del mon tal cura

\numero{10} que noy jaquis / de terribles affanys

\numero{11} perdues son / en nombre mes dels guanys

\numero{12} en'sguart del ver / lo seny es oradura.

\numero{13} el hom girat / al mon d'auol saber

\numero{14} Ydoles creu / donant y plena fer

\numero{15} en bens mundans / creu esser complit be

\numero{16} volent morir / solament te d'hauer.

\end{estrofa}



\begin{estrofa}

\numero{17} Si algu pogues / de fortuna tener

\numero{18} ab vn fort clau / sa roda quant es sus,

\numero{19} fora li bo / anar amunt e jus,

\numero{20} soferint mal / per trobar tal mester.

\numero{21} [34r] mas los no tals / no han tanta vigor

\numero{22} que tant gran pes / ab lur for\c{c}a retingen

\numero{23} ne'gynis nos fan / que d'alt e baix no vingen

\numero{24} los quiseuran / en la falsa honor.

\end{estrofa}



\begin{estrofa}

\numero{25} Per nengun temps / perdra fortuna'rdor

\numero{26} de fer lochs plans / deles timpes e munt

\numero{27} e quant del tot / son desijat es junt

\numero{28} e no pot fer / pus ab sa gran furor,

\numero{29} torna refer / deshabitada casa

\numero{30} que'n poch spay / hauie feta buyda

\numero{31} el'hom desert / que'nriquir james cuyda

\numero{32} vmple graners / olis e vins embasa.

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat / no deu portar spasa

\numero{34} com afoll hom / deu franch arbitre pendre

\numero{35} qui no creu cert / que fortuna pot pendre

\numero{36} los bens mouents / y los sehents arrasa.

\numero{37} mas l'hom del mon / per fortuna mirat

\numero{38} ab lo'sguart / de riallosa cara

\numero{39} la part de plor / no ha vista encara

\numero{40} creu que sos ginys / l'han be prosperat.

\end{estrofa}
