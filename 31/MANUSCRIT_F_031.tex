\poema{(F 31) [31]}\\

\begin{rubrica}

Altra

\end{rubrica}
 
\end{estrofa}


\begin{estrofa}

\numero{1} Molts homens hoig clamar se de fortu\textit{n}a

\numero{2} En malahir aquella per sos actes

\numero{3} Volents ab ley fer amjgables pactes

\numero{4} Donant los be e que tostems fos vna

\numero{5} No recordant sa propria natura

\numero{6} Ques lalt baxar e lo baix muntar alt

\numero{7} E quj daco se dona gran desalt

\numero{8} Lexe los bens portants ab si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} A deu no plach hauer del mon tal cura

\numero{10} Que noy jaqujs de terribles afanys

\numero{11} Perdues son en nombre mes quels guanys

\numero{12} Nesguart del uer lo seny es oradura

\numero{13} E lom girat al mundanal saber

\numero{14} Ydoles creu donant hy plena fe

\numero{15} En bens mundans creu esser complit be

\numero{16} Volent morir solament per hauer

\end{estrofa}



\begin{estrofa}

\numero{17} [31r] Salgu pogues de fortuna tener

\numero{18} Ab hun fort clau sa roda qua\textit{n}t es sus

\numero{19} Ffora li bo anar amunt e jus

\numero{20} Soferint mals per trobar tal mester

\numero{21} Mas los metalls no han tanta vigor

\numero{22} Que tan gran pes ab lur forca retjnguen

\numero{23} He ginys no fan que dalt en baix no vinguen

\numero{24} Los quj seuran en la falssa honor

\end{estrofa}



\begin{estrofa}

\numero{25} Per negun temps perdra fortunador

\numero{26} De fer lochs plans deles timpes\footnote{\textit{Titlla sup\`{e}rflua damunt la
consonant} p.} e mu\interlin{n}t

\numero{27} E quant del tot son desijat es junt

\numero{28} E no pot fer p\textit{us }ab sa gran furor

\numero{29} Torna reffer desabitada casa

\numero{30} Quan poch espay hauia feta buyda

\numero{31} E lom desert e quen requjr james cuyda

\numero{32} Omple graners olis e vins enbasa

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat no deu portat espasa

\numero{34} Com afoll hom deu franch arbitre perdre

\numero{35} Quj no creu cert que fortuna pot perdre

\numero{36} Los bens mouents e los sientsarrasa

\numero{37} Mas lom del mon per fortuna mjrat

\numero{38} Ab lo esguart de riallosa cara

\numero{39} La part del plor no ha vista encara

\numero{40} Creu que sos ginys lan de bens prosperat

\end{estrofa}



\begin{estrofa}

\numero{41} [31v] Lir entre carts propi so comparat

\numero{42} A tantalus per continuu desig

\numero{43} Nose per que tots dies hi affig

\numero{44} Puys que mes prop compte desesperat

\end{estrofa}
