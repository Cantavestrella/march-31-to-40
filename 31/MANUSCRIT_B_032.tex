\begin{estrofa}

\numero{1} Molts homens hotx clamarse de fortuna

\numero{2} E malehir aq\textit{ue}lla per sos actes

\numero{3} Volents fermar ab ella vns tals pactes

\numero{4} Quels donas be per tostemps essent vna

\numero{5} No recordants sa propia natura

\numero{6} Ques lalt baxar e lo bax mu\textit{n}tar alt

\numero{7} E qui da\c{c}o se dona gran desalt

\numero{8} Lexe los bens portans ab si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} A deu no plach hauer del mo\textit{n} tal cura

\numero{10} Que noy jaquis menys delits q\textit{ue} affanys

\numero{11} Per dues son en nombre mes quels gua\textit{n}ys

\numero{12} Nes sguart del ver lo seny es oradura

\numero{13} Elom girat al mundanal saber

\numero{14} Ydolas creu donant y plena fe

\numero{15} En bens mundans creu esser co\textit{m}plit be

\numero{16} Volent morir solament per hauer

\end{estrofa}



\begin{estrofa}

\numero{17} Si algu pusques de fortuna tener

\numero{18} Ab vn fort clau sa roda qua\textit{n}t es sus

\numero{19} Forali bo anar amu\textit{n}t e jus

\numero{20} Soferi\textit{n}t mals per trobar son mester

\numero{21} Mas los matalls no han tanta vigor

\numero{22} Que tan gran pes ab llur for\c{c}a retinguen

\numero{23} [28r] E ginys no fan que dalt en bax no vingue\textit{n}

\numero{24} Los qui seia en la fal\c{c}a honor

\end{estrofa}



\begin{estrofa}

\numero{25} Per nengun temps perdra fortunardor

\numero{26} De fer lochs plans deles cumbes e munt

\numero{27} E qua\textit{n}t d\textit{e}l tot son desijat es junt

\numero{28} Eno pot fer pus ab sa gran furor

\numero{29} Torna afer destibitada casa

\numero{30} Quen molt breu temps hauia feta buyda

\numero{31} EL pobre hom quen riquir ja mes cuyda

\numero{32} Vmpla granes olis e vins en vasa

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat no deu portar spasa

\numero{34} Com afoll hom deu franch arbitrespendre

\numero{35} Qui no creu cert que fortuna pot perdre

\numero{36} Los bens mouents e los sehents arasa

\numero{37} Mas lom d\textit{e}l mon per fortuna mirat

\numero{38} Ab lo sguart de riallosa cara

\numero{39} La part d\textit{e}l plor no ha vista encara

\numero{40} Creu que sos ginys lan de bens prosperat

\end{estrofa}
