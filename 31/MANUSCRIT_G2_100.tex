\begin{estrofa}

\numero{1} Molts homens oig / clamarse defortuna

\numero{2} emalair aquella per sos actes

\numero{3} volent ab lleys / fer amigables pactes

\numero{4} donant los be / eque tots temps fos huna

\numero{5} No rrecordant / sap\textit{ro}pia natura

\numero{6} ques lalt baxar / yl baix mu\textit{n}tar alt

\numero{7} equida\c{c}o sedona / gran desalt

\numero{8} lexe los bens portants absi fretura

\end{estrofa}



\begin{estrofa}

\numero{9} A deu noplach / auer del mon tal cura

\numero{10} que noy jaquis / de terribles afanys

\numero{11} per dues son / en nombre mes quels guanys

\numero{12} nesguart del ver / lo Seny es horadura

\numero{13} Elomgirat / al mundanal saber

\numero{14} ydoles creu / donanthi plena fe

\numero{15} quenbens mundans / creu ess\textit{er} complit be

\numero{16} volent morir / solament per auer

\end{estrofa}



\begin{estrofa}

\numero{17} Salgu pos ques / de fortuna tener

\numero{18} ab vn fort clau / sarodaqua\textit{n}t es sus

\numero{19} fora libo / anar amunt y jus

\numero{20} soferint mals / per trobar tal mester

\numero{21} Mas los metalls / no antanta vigor

\numero{22} quetangranpes / abllurfor\c{c}a retinguen

\numero{23} neginys nearts / no faran q\textit{ue} no vinguen

\numero{24} los qui seuran / enla fal\c{c}a honor

\end{estrofa}



\begin{estrofa}

\numero{25} [139r] Per algun temps / perdra fortunardor

\numero{26} defer llochs plans / deles timpes emunt

\numero{27} e quant deltot / son desijat esjunt

\numero{28} eno pot fer pus / ab sagran furor

\numero{29} Torna refer desabitada casa

\numero{30} quenpoch spay / auia feta buyda

\numero{31} e lom desert quen rriquir james cuyda

\numero{32} vmplegraners / olis evins enbasa

\end{estrofa}



\begin{estrofa}

\numero{33} Enson costat / no deu portar spasa

\numero{34} com afoll hom deu / franch arbitre perdre

\numero{35} qui no creu cert / que fortunapot perdre

\numero{36} los bens mouents / e los scients arrasa

\numero{37} Mas lom del mon / per fortuna mirat

\numero{38} abriallosa cara\footnote{\textit{Seq\"{u}\`{e}ncia expuntuada}.} / ab lo esguart
deriallosacara

\numero{39} lapart del plor / no avista encara

\numero{40} creu quesos ginys / lan debens p\textit{ro}sperat

\end{estrofa}
