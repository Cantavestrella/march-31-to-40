\begin{estrofa}

\numero{1} \textparagraph\  Molts homens hog clamarse de fortuna

\numero{2} En maleyr aquela per sos actes

\numero{3} Volents fermar ab ella cons tals pactes

\numero{4} Quels donas be eque tostems fos vna

\numero{5} No recordans sa propia natura

\numero{6} Que es lalt baxar elo bax muntar alt

\numero{7} E quj daco se dona gran desalt

\numero{8} Lexe los bens portants ab si fretura

\end{estrofa}



\begin{estrofa}

\numero{9} A deu no plach auer del mon tal cura

\numero{10} Que noy jaqujs de terribles afanys

\numero{11} Perdues son ennombre mes quels guayns

\numero{12} Nesguart del ver lo seny es oradura

\numero{13} En lom girat almundanal saber

\numero{14} Ydoles creu donanthi plena fe

\numero{15} Enbens mundans creu esser complit be

\numero{16} Volent morir solament per auer

\end{estrofa}



\begin{estrofa}

\numero{17} [61] Si algu pogues de fortuna tener

\numero{18} Ab hun fort clau sa roda quant es sus

\numero{19} Fforalj bo anant amunt ejus

\numero{20} Soffarint mals per trobar tal mester

\numero{21} Mas los matalls nohan tanta vigor

\numero{22} Que ablur pes tangran forca retinguen

\numero{23} He gin\textit{y}s no fan que dalt en bax no vinguen

\numero{24} Los quj seuran enla falsa honor

\end{estrofa}



\begin{estrofa}

\numero{25} Pernegun temps perdra fortunador

\numero{26} De fer lochs plans deles timpes emunts

\numero{27} E quant del tot son desijat es junt

\numero{28} E no pot fer p\textit{us }ab sa gran furor

\numero{29} Torna refer desabitada cassa

\numero{30} Quen poch espay hauja feta buyda

\numero{31} E lom desert quen requir james cuyda

\numero{32} Humple granes holis e vins enbasa

\end{estrofa}



\begin{estrofa}

\numero{33} En son costat no deu portat espasa

\numero{34} Com afoll home deu franch arbitre pendre

\numero{35} Quj no creu cert que fortuna pot perdre

\numero{36} Los bens mouents elos sients arasa

\numero{37} Mas lom del mon per fortuna mjrat

\numero{38} Ab lo esguart \sout{de rias} de rialosa cara

\numero{39} Lapart del plor noha vista encara

\numero{40} Creuque sos gjn\textit{y}s lan debens prosperat

\end{estrofa}
